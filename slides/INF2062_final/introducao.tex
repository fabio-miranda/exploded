\begin{frame}\frametitle{\emph{Smart visibility}}
\begin{itemize}
	\item Uma "visibilidade inteligente" leva em consideração:
	\begin{itemize}
		\item A \textbf{relevância dos objetos e suas características}, e não apenas o seu posicionamento no espaço. Um objeto importante pode ser visualizado apesar de estar sendo obstruído por algum outro objeto mais próximo do observador.
		\item A \textbf{familiaridade do observador com os objetos}. A partir de um objeto parcialmente visível, o observador pode completá-lo mentalmente, com base em sua experiência.
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}\frametitle{Vistas explodidas}
\begin{itemize}
	\item Vistas explodidas modificam o posicionamento espacial dos componentes de um objeto para que os mais "interessantes" estejam visíveis.
	\item Permitem o entendimento da estrutura global do objeto e a relação espacial entre os componentes.
\end{itemize}
