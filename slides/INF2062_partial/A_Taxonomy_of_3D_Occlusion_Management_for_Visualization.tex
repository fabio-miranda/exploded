\section{Características da Vista Explodida}
\begin{frame}\frametitle{\emph{A Taxonomy of 3D Occlusion Management for Visualization} \cite{1446259}} 
\begin{itemize}
	\item Apresenta uma classificação das técnicas de gerenciamento de oclusão, sendo possível perceber de forma objetiva as diferenças entre as diversas técnicas.
	\item A partir de 25 características, o trabalho deriva 5 \emph{patterns} para reduzir a oclusão em modelos 3D.
\end{itemize}
\end{frame}

\begin{frame}\frametitle{Características das técnicas de gerenciamento de oclusão (1)}
\begin{itemize}
	\item \textbf{Propósito principal}:
	\begin{itemize}
		\item \textbf{Descoberta}: tem como objetivo fazer com que o usuário fique ciente de partes obstruídas, não necessariamente deixando mais acessível as informações do objeto.
		\item \textbf{Acesso}: diferentemente da \textbf{descoberta}, deixa necessariamente as informações do objeto mais acessíveis.
		\item \textbf{Relacionamento espacial}: busca deixar o contexto em que o objeto está inserido visível e mais fácil de ser entendido pelo usuário. 
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}\frametitle{Características das técnicas de gerenciamento de oclusão (3)}
\begin{itemize}
	\item \textbf{Sugestões de profundidade (\emph{depth cues})}: o nível de percepção de profundidade que determinada técnica transmite ao observador.
	\begin{itemize}
		\item \textbf{Baixa}.
		\item \textbf{Ligeiramente baixa}.
		\item \textbf{Média}.
		\item \textbf{Ligeiramente alta}.
		\item \textbf{Alta}.
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}\frametitle{Características das técnicas de gerenciamento de oclusão (5)}
\begin{itemize}
	\item \textbf{Modelo de interação}: como se dá a interação com o usuário.
	\begin{itemize}
		\item \textbf{Passivo}: sem intervenção do usuário.
		\item \textbf{Híbrido}: pode haver uma transição entre o modo passivo e o ativo.
		\item \textbf{Ativo}: intervensão ativa do usuário.
	\end{itemize}
\end{itemize}
\end{frame}
	
\begin{frame}\frametitle{Características das técnicas de gerenciamento de oclusão (6)}
\begin{itemize}
	\item \textbf{Invariantes do objeto}:
	\begin{itemize}
		\item \textbf{Posição e orientação}: posição e orientação.
		\item \textbf{Geometria}: forma e tamanho do objeto.
		\item \textbf{Aparência}: cor, textura e material.
	\end{itemize}
\end{itemize}
\end{frame}
