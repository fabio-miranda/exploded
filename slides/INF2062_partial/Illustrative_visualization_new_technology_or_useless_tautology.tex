\subsection{Illustrative visualization: new technology or useless tautology?}
\begin{frame}\frametitle{\emph{Illustrative visualization: new technology or useless tautology?} \cite{1408633}} 
\begin{itemize}
	\item Visualização ilustrativa: Representação visual interativa e expressiva, assistida por computador, motivada por ilustrações tradicionais. Dois níveis de abstração:
	\begin{itemize}
		\item Abstrações de baixo nível: \textbf{como} renderizar as características de interesse (ex: \emph{toon shading}).
		\item Abstrações de alto nível: \textbf{o que} renderizar. (ex: \emph{cutaways} e vistas explodidas).
				\begin{itemize}
					\item Necessitam de informações sobre a relevância dos dados.
					\item \emph{Smart visibility techniques} \cite{Viola-05-Smart}.
				\end{itemize}
	\end{itemize}
\end{itemize}	
\end{frame}
